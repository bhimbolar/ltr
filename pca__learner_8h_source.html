00168 template <typename TElement>
00169 void PCALearner<TElement>::learnImpl(const DataSet<TElement>& data_set,
00170                                      LinearConverter* linear_converter) {
00171   VectorXd eigen_average_features;
00172   getFeaturesAverageValues(data_set, &eigen_average_features);
00173 
00174   MatrixXd source_matrix = data_set.get_features_matrix();
00175 
00176   for (int object_index = 0;
00177        object_index < source_matrix.cols();
00178        ++object_index) {
00179     source_matrix.col(object_index) =
00180       source_matrix.col(object_index) - eigen_average_features;
00181   }
00182 
00183   JacobiSVD<MatrixXd> jacobi_svd = source_matrix.jacobiSvd(ComputeThinU);
00184   MatrixXd matrix_U = jacobi_svd.matrixU();
00185   singular_values_ = jacobi_svd.singularValues();
00186   int component_count = rule_->getNumberOfComponents(singular_values_);
00187 
00188   matrix_U.conservativeResize(matrix_U.rows(), component_count);
00189   linear_converter->set_factor(matrix_U.transpose());
00190 
00191   VectorXd shift = VectorXd::Zero(component_count);
00192   linear_converter->set_shift(shift);
00193 }
