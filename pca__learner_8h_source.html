00172 template <typename TElement>
00173 void PCALearner<TElement>::learnImpl(const DataSet<TElement>& data_set,
00174                                      LinearConverter* linear_converter) {
00175   Features average_features;
00176   getFeaturesAverageValues(data_set, &average_features);
00177   VectorXd eigen_average_features = StdVectorToEigenVector(average_features);
00178 
00179   MatrixXd source_matrix = DataSetToEigenMatrix(data_set).transpose();
00180 
00181   for (int object_index = 0;
00182        object_index < source_matrix.cols();
00183        ++object_index) {
00184     source_matrix.col(object_index) =
00185       source_matrix.col(object_index) - eigen_average_features;
00186   }
00187 
00188   JacobiSVD<MatrixXd> jacobi_svd = source_matrix.jacobiSvd(ComputeThinU);
00189   MatrixXd matrix_U = jacobi_svd.matrixU();
00190   singular_values_ = jacobi_svd.singularValues();
00191   int component_count = rule_->getNumberOfComponents(singular_values_);
00192 
00193   matrix_U.conservativeResize(matrix_U.rows(), component_count);
00194   linear_converter->set_factor(matrix_U.transpose());
00195 
00196   VectorXd shift(component_count);
00197   InitEigenVector(&shift);
00198   linear_converter->set_shift(shift);
00199 }
