00074                                      LinearConverter* linear_converter) {
00075   Features average_features;
00076   getFeaturesAverageValues(data_set, &average_features);
00077   VectorXd eigen_average_features = StdVectorToEigenVector(average_features);
00078 
00079   MatrixXd source_matrix = DataSetToEigenMatrix(data_set).transpose();
00080 
00081   for (int object_index = 0;
00082        object_index < source_matrix.cols();
00083        ++object_index) {
00084     source_matrix.col(object_index) =
00085       source_matrix.col(object_index) - eigen_average_features;
00086   }
00087 
00088   JacobiSVD<MatrixXd> jacobi_svd = source_matrix.jacobiSvd(ComputeThinU);
00089   MatrixXd matrix_U = jacobi_svd.matrixU();
00090   VectorXd singular_values = jacobi_svd.singularValues();
00091   int component_count = kaiserRule(singular_values);
00092 
00093   matrix_U.conservativeResize(matrix_U.rows(), component_count);
00094   linear_converter->set_factor(matrix_U.transpose());
00095 
00096   VectorXd shift(component_count);
00097   InitEigenVector(&shift);
00098   linear_converter->set_shift(shift);
