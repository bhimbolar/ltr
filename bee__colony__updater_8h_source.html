00095                                      random_point[dim_index]);
00096     if (this->set_->isInside(new_point) &&
00097         this->function_->computeValue(new_point) <
00098         this->function_->computeValue(it.point())) {
00099       population->updatePoint(it.point_id(), new_point);
00100     }
00101     sum_value += this->function_->computeValue(it.point());
00102     ++current_point_index;
00103   }
00104   for (Population::Iterator it = population->begin();
00105        it != population->end();
00106        ++it) {
00107     // probability to reject point
00108     double probability = this->function_->computeValue(it.point()) / sum_value;
00109     if (randomizer.doubleRand(0, 1) <= probability) {
