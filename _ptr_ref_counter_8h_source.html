00034 //      PtrRefCounter:  Object used to keep track of the reference count for a group
00035 //                                      of smart pointers pointing to the same object. When the last
00036 //                                      Active reference is removed, the Raw Object that is pointed
00037 //                                      to should be deleted by the smart pointer. When the last
00038 //                                      Reference is removed (either active or inactive), the
00039 //                                      Reference Counting Object (this instantiated class) should
00040 //                                      be destroyed by the smart pointer

00080 //
00081 // description:
00082 //              thread safe, adds an active reference to the count as long as the active
00083 //              count is not zero
00084 // return:
00085 //              true if sucessful;
00086 //              false if the Reference Counter Object and Raw pointer (from the calling
00087 //              Smart Pointer) are no longer valid (i.e. should be set to NULL)
00088 inline bool PtrRefCounter::AddActiveRef( void )
00089 {

00092         // this mutex ensures that once the active count is zero, it will

00139 //      SubtractActiveRef  [public]
00140 //
00141 // description:
00142 //              thread safe, subtracts an active reference from the count
00143 //              NOTE: a call to this function automatically invalidates the pointer that
00144 //              was previously pointing to this object - it should immediately be set
00145 //              to NULL
00146 // return:
00147 //              0 if nothing needs to be done;
00148 //              non-zero if the caller should delete the Raw object it is pointing to;
00149 //              DELETE_REF_COUNTER if the caller should delete the Reference Counter
00150 //              Object (DELETE_REF_COUNTER is a non-zero number, so if it is returned,
00151 //              obviously, both objects should be deleted)
00152 inline int PtrRefCounter::SubtractActiveRef( void )
00153 {
00154         int iFunctionRetVal = 0;                // functions value that will be returned
00155         m_pMutex.Lock();
00156         {
00157                 --m_uActiveCount;
00158                 if( ShouldThisObjectBeDeleted() )
00159                 {
00160                         // no more active or inactive references, so both objects should
00161                         // be deleted
00162                         iFunctionRetVal = DELETE_REF_COUNTER;
00163                 }
00164                 else if( m_uActiveCount == 0)
00165                 {
00166                         // no more active counts, but at least one inactive count remains,
00167                         // so delete the raw pointer, but not the Reference Counting Object
00168                         iFunctionRetVal = -1;   // return a non-zero number
00169                 }
00170         }
00171         m_pMutex.Unlock();
00172 
00173         return( iFunctionRetVal );
00174 }
