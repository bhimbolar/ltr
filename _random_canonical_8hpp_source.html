00475     template<typename RealType, int prec> RealType FixedC() throw() {
00476       // A real of type RealType in [0, 1] with precision prec
00477       STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
00478                     std::numeric_limits<RealType>::radix == 2,
00479                     "FixedC(): bad real type RealType");
00480       STATIC_ASSERT(prec > 0 && prec <= std::numeric_limits<RealType>::digits,
00481                     "FixedC(): invalid precision");
00482       if (prec < width) {
00483         // Sample an integer in [0, n) where n = 2^prec + 1.  This uses the
00484         // same logic as Unsigned(n - 1).  However, unlike Unsigned, there
00485         // doesn't seem to be much of a penalty for the 64-bit arithmetic here
00486         // when result_type = unsigned long long.  Presumably this is because
00487         // the compiler can do some of the arithmetic.
00488         const result_type
00489           n = (result_type(1) << (prec < width ? prec : 0)) + 1,
00490           // Computing this instead of 2^width/n suffices, because of the form
00491           // of n.
00492           r = Generator::mask / n,
00493           m = r * n;
00494         result_type u;
00495         do
00496           u = Generator::Ran();
00497         while (u >= m);
00498         // u is rv in [0, r * n)
00499         return RandomPower2::shiftf<RealType>(RealType(u / r), -prec);
00500         // Could also special case prec < 64, using Ran64().  However the
00501         // general code below is faster.
00502       } else {                  // prec >= width
00503         // Synthesize a prec+1 bit random, Y, width bits at a time.  If number
00504         // is odd, return Fixed<RealType, prec>() (w prob 1/2); else if number
00505         // is zero, return 1 (w prob 1/2^(prec+1)); else repeat.  Normalizing
00506         // probabilities on returned results we find that Fixed<RealType,
00507         // prec>() is returned with prob 2^prec/(2^prec+1), and 1 is return
00508         // with prob 1/(2^prec+1), as required.  Loop executed twice on average
00509         // and so consumes 2rvs more than rvs for Fixed<RealType, prec>().  As
00510         // in FloatZ, do NOT try to save on calls to Ran() by using the
00511         // leftover bits from Fixed.
00512         while (true) {
00513           // If prec + 1 < width then mask x with (1 << prec + 1) - 1
00514           const result_type x = Generator::Ran(); // Low width bits of Y
00515           if (x & 1u)                             // Y odd?
00516             return Fixed<RealType, prec>(); // Prob 1/2 on each loop iteration
00517           if (x)
00518             continue;               // Y nonzero
00519           int s = prec + 1 - width; // Bits left to check (s >= 0)
00520           while (true) {
00521             if (s <= 0)         // We're done.  Y = 0
00522               // Prob 1/2^(prec+1) on each loop iteration
00523               return RealType(1); // We get here once every 60000 yrs (p = 64)!
00524             // Check the next min(s, width) bits.
00525             if (Generator::Ran() >> (s > width ? 0 : width - s))
00526               break;
00527             s -= width;         // Decrement s
00528           }
00529         }
00530       }
00531     }

00871   template<class Generator> template<typename UIntT>
00872   inline typename UIntT::type
00873   RandomCanonical<Generator>::Unsigned(typename UIntT::type n) throw() {
00874     // A random unsigned in [0, n].  In n fits in 32-bits, call with UIntType =
00875     // u32 and onep = true else call with UIntType = u64 and onep = false.
00876     // There are a few cases (e.g., n = 0x80000000) where on a 64-bit machine
00877     // with a 64-bit Generator it would be quicker to call this with UIntType =
00878     // result_type and invoke Ran().  However this speed advantage disappears
00879     // if the argument isn't a compile time constant.
00880     //
00881     // Special case n == 0 is handled by the callers of Unsigned.  The
00882     // following is to guard against a division by 0 in the return statement
00883     // (but it shouldn't happen).
00884     n = n ? n : 1U;             // n >= 1
00885     // n1 = n + 1, but replace overflowed value by 1.  Overflow occurs, e.g.,
00886     // when n = u32::mask and then we have r1 = 0, m = u32::mask.
00887     const typename UIntT::type n1 = ~n ? n + 1U : 1U;
00888     // "Ratio method".  Find m = r * n1 - 1, s.t., 0 < (q - n1 ) < m <= q,
00889     // where q = max(UIntType), and sample in u in [0, m] and return u / r.  If
00890     // onep then we use Ran32() else Rand64().
00891     const typename UIntT::type
00892       // r = floor((q + 1)/n1), r1 = r - 1, avoiding overflow.  Actually
00893       // overflow can occur if std::numeric_limits<u32>::digits == 64, because
00894       // then we can have onep && n > U32_MASK.  This is however ruled out by
00895       // the callers to Unsigned.  (If Unsigned is called in this way, the
00896       // results are bogus, but there is no error condition.)
00897       r1 = ((UIntT::width == 32 ? typename UIntT::type(u32::mask) :
00898              typename UIntT::type(u64::mask)) - n) / n1,
00899       m = r1 * n1 + n;          // m = r * n1 - 1, avoiding overflow
00900     // Here r1 in [0, (q-1)/2], m in [(q+1)/2, q]
00901     typename UIntT::type u;     // Find a random number in [0, m]
00902     do
00903       // For small n1, this is executed once (since m is nearly q).  In the
00904       // worst case the loop is executed slightly less than twice on average.
00905       u = UIntT::width == 32 ? typename UIntT::type(Generator::Ran32()) :
00906         typename UIntT::type(Generator::Ran64());
00907     while (u > m);
00908     // Now u is in [0, m] = [0, r * n1), so u / r is in [0, n1) = [0, n].  An
00909     // alternative unbiased method would be u % n1; but / appears to be faster.
00910     return u / (r1 + 1U);
00911   }

00971   RealType RandomCanonical<Generator>::FloatZ(int b, result_type m) throw() {

00973     // usable random bits.
00974     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
00975                   std::numeric_limits<RealType>::radix == 2,
00976                   "FloatZ: bad real type RealType");
00977     STATIC_ASSERT(prec > 0 && prec <= std::numeric_limits<RealType>::digits,
00978                   "FloatZ: invalid precision");
00979     STATIC_ASSERT(erange >= 0, "FloatZ: invalid exponent range");
00980     // With subnormals: condition that smallest number is representable
00981     STATIC_ASSERT(!RANDOMLIB_HASDENORM(RealType) ||
00982                   // Need 1/2^(erange+prec) > 0
00983                   prec + erange <= std::numeric_limits<RealType>::digits -
00984                   std::numeric_limits<RealType>::min_exponent,
00985                   "FloatZ: smallest number cannot be represented");
00986     // Without subnormals :condition for no underflow in while loop
00987     STATIC_ASSERT(RANDOMLIB_HASDENORM(RealType) ||
00988                   // Need 1/2^(erange+1) > 0
00989                   erange <= - std::numeric_limits<RealType>::min_exponent,
00990                   "FloatZ: underflow possible");
00991 
00992     // Simpler (but slower) version of FloatZ.  However this method cannot
00993     // handle the full range of exponents and, in addition, is slower on
00994     // average.
00995     // template<typename RealType, int prec, int erange, bool up>
00996     // RealType FloatZ() {
00997     //   RealType x = Fixed<RealType, erange + 1>();
00998     //   int s;         // Determine exponent (-erange <= s <= 0)
00999     //   frexp(x, &s);      // Prob(s) = 2^(s-1)
01000     //   // Scale number in [1,2) by 2^(s-1).  If x == 0 scale number in [0,1).
01001     //   return ((up ? FixedU<RealType, prec - 1>() :
01002     //            Fixed<RealType, prec - 1>()) + (x ? 1 : 0)) *
01003     //     RandomPower2::pow2<RealType>(s - 1);
01004     // }
01005     //
01006     // Use {a, b} to denote the inteval: up ? (a, b] : [a, b)
01007     //
01008     // The code produces the number as
01009     //
01010     // Interval             count       prob = spacing
01011     // {1,2} / 2            2^(prec-1)  1/2^prec
01012     // {1,2} / 2^s          2^(prec-1)  1/2^(prec+s-1)      for s = 2..erange+1
01013     // {0,1} / 2^(erange+1) 2^(prec-1)  1/2^(prec+erange)
01014 
01015     // Generate prec bits in {0, 1}
01016     RealType x = up ? FixedU<RealType, prec>() : Fixed<RealType, prec>();
01017     // Use whole interval if erange == 0 and handle the interval {1/2, 1}
01018     if (erange == 0 || (up ? x > RealType(0.5) : x >= RealType(0.5)))
01019       return x;
01020     x += RealType(0.5);         // Shift remaining portion to {1/2, 1}
01021     if (b == 0) {
01022       m = Generator::Ran();     // Random bits
01023       b = width;                // Bits available in m
01024     }
01025     int sm = erange;            // sm = erange - s + 2
01026     // Here x in {1, 2} / 2, prob 1/2
01027     do {                        // s = 2 thru erange+1, sm = erange thru 1
01028       x /= 2;
01029       if (m & 1u)
01030         return x;               // x in {1, 2} / 2^s, prob 1/2^s
01031       if (--b)
01032         m >>= 1;
01033       else {
01034         m = Generator::Ran();
01035         b = width;
01036       }
01037     } while (--sm);
01038     // x in {1, 2} / 2^(erange+1), prob 1/2^(erange+1).  Don't worry about the
01039     // possible overhead of the calls to pow here.  We rarely get here.
01040     if (RANDOMLIB_HASDENORM(RealType) || // subnormals allowed
01041         // No subnormals but smallest number still representable
01042         prec + erange <= -std::numeric_limits<RealType>::min_exponent + 1 ||
01043         // Possibility of underflow, so have to test on x.  Here, we have -prec
01044         // + 1 < erange + min_exp <= 0 so pow2 can be used
01045         x >= (RealType(1) +
01046               RandomPower2::pow2<RealType>
01047               (erange + std::numeric_limits<RealType>::min_exponent)) *
01048         (erange + 1 > -RandomPower2::minpow ?
01049          std::pow(RealType(2), - erange - 1) :
01050          RandomPower2::pow2<RealType>(- erange - 1)))
01051       // shift x to {0, 1} / 2^(erange+1)
01052       // Use product of pow's since max(erange + 1) =
01053       // std::numeric_limits<RealType>::digits -
01054       // std::numeric_limits<RealType>::min_exponent and pow may underflow
01055       return x -
01056         (erange + 1 > -RandomPower2::minpow ?
01057          std::pow(RealType(2), -(erange + 1)/2) *
01058          std::pow(RealType(2), -(erange + 1) + (erange + 1)/2) :
01059          RandomPower2::pow2<RealType>(- erange - 1));
01060     else
01061       return up ?               // Underflow to up ? min() : 0
01062         // pow is OK here.
01063         std::pow(RealType(2), std::numeric_limits<RealType>::min_exponent - 1)
01064         : RealType(0);

01078   // True with probability p.  true if p >= 1, false if p <= 0 or isnan(p).
01079   template<class Generator> template<typename RealType>
01080   inline bool RandomCanonical<Generator>::ProbF(RealType p) throw() {
01081     // Simulate Float<RealType>() < p.  The definition involves < (instead of
01082     // <=) because Float<RealType>() is in [0,1) so it is "biased downwards".
01083     // Instead of calling Float<RealType>(), we generate only as many bits as
01084     // necessary to determine the result.  This makes the routine considerably
01085     // faster than Float<RealType>() < x even for type float.  Compared with
01086     // the inexact Fixed<RealType>() < p, this is about 20% slower with floats
01087     // and 20% faster with doubles and long doubles.
01088     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
01089                   std::numeric_limits<RealType>::radix == 2,
01090                   "ProbF(p): invalid real type RealType");
01091     // Generate Real() c bits at a time where c is chosen so that cast doesn't
01092     // loose any bits and so that it uses up just one rv.
01093     const int c = std::numeric_limits<RealType>::digits > width ?
01094       width : std::numeric_limits<RealType>::digits;
01095     STATIC_ASSERT(c > 0, "ProbF(p): Illegal chunk size");
01096     const RealType mult = RandomPower2::pow2<RealType>(c);
01097     // A recursive definition:
01098     //
01099     // return p > RealType(0) &&
01100     //   (p >= RealType(1) ||
01101     //    ProbF(mult * p - RealType(Integer<result_type, c>())));
01102     //
01103     // Pre-loop tests needed to avoid overflow
01104     if (!(p > RealType(0)))     // Ensure false if isnan(p)
01105       return false;
01106     else if (p >= RealType(1))
01107       return true;
01108     do {                        // Loop executed slightly more than once.
01109       // Here p is in (0,1).  Write Fixed() = (X + y)/mult where X is an
01110       // integer in [0, mult) and y is a real in [0,1).  Then Fixed() < p
01111       // becomes p' > y where p' = p * mult - X.
01112       p *= mult;                // Form p'.  Multiplication is exact
01113       p -= RealType(Integer<result_type, c>()); // Also exact
01114       if (p <= RealType(0))
01115         return false;           // If p' <= 0 the result is definitely false.
01116       // Exit if p' >= 1; the result is definitely true.  Otherwise p' is in
01117       // (0,1) and the result is true with probability p'.
01118     } while (p < RealType(1));
01119     return true;
01120   }

01133   // True with probability x/y (ratio of reals)
01134   template<class Generator> template<typename RealType>
01135   inline bool RandomCanonical<Generator>::ProbF(RealType x, RealType y)
01136     throw() {
01137     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
01138                   std::numeric_limits<RealType>::radix == 2,
01139                   "ProbF(x,y): invalid real type RealType");
01140     if (!(x > RealType(0) && y >= RealType(0))) // Do the trivial cases
01141       return false;             // Also if either x or y is a nan
01142     else if (x >= y)
01143       return true;
01144     // Now 0 < x < y
01145     int ex, ey;                 // Extract exponents
01146     x = std::frexp(x, &ex);
01147     y = std::frexp(y, &ey);
01148     // Now 0.5 <= x,y < 1
01149     if (x > y) {
01150       x *= RealType(0.5);
01151       ++ex;
01152     }
01153     int s = ey - ex;
01154     // Now 0.25 < x < y < 1, s >= 0, 0.5 < x/y <= 1
01155     // Return true with prob 2^-s * x/y
01156     while (s > 0) {             // With prob 1 - 2^-s return false
01157       // Check the next min(s, width) bits.
01158       if (Generator::Ran() >> (s > width ? 0 : width - s))
01159         return false;
01160       s -= width;
01161     }
01162     // Here with prob 2^-s
01163     const int c = std::numeric_limits<RealType>::digits > width ?
01164       width : std::numeric_limits<RealType>::digits;
01165     STATIC_ASSERT(c > 0, "ProbF(x,y): invalid chunk size");
01166     const RealType mult = RandomPower2::pow2<RealType>(c);
01167     // Generate infinite precision z = Real().
01168     // As soon as we know z > y, start again
01169     // As soon as we know z < x, return true
01170     // As soon as we know x < z < y, return false
01171     while (true) {              // Loop executed 1/x on average
01172       double xa = x, ya = y;
01173       while (true) {            // Loop executed slightly more than once
01174         // xa <= ya, ya > 0, xa < 1.
01175         // Here (xa,ya) are in (0,1).  Write Fixed() = (X + y)/mult where X is
01176         // an integer in [0, mult) and y is a real in [0,1).  Then Fixed() < z
01177         // becomes z' > y where z' = z * mult - X.
01178         const RealType d = RealType(Integer<result_type, c>());
01179         if (ya < RealType(1)) {
01180           ya *= mult;           // Form ya'
01181           ya -= d;
01182           if (ya <= RealType(0))
01183             break;              // z > y, start again
01184         }
01185         if (xa > RealType(0)) {
01186           xa *= mult;           // From xa'
01187           xa -= d;
01188           if (xa >= RealType(1))
01189             return true;        // z < x
01190         }
01191         if (xa <= RealType(0) && ya >= RealType(1))
01192           return false;         // x < z < y
01193       }
01194     }
01195   }

01198   inline std::bitset<nbits> RandomCanonical<Generator>::Bits() throw() {
01199     // Return nbits random bits
01200     STATIC_ASSERT(nbits >= 0, "Bits<n>(): invalid nbits");
01201     const int ulbits = std::numeric_limits<bitset_uint_t>::digits;
01202     STATIC_ASSERT(2 * ulbits >= width,
01203                   "Bits<n>(): integer constructor type too narrow");
01204     std::bitset<nbits> b;
01205     int m = nbits;
01206 
01207     while (m > 0) {
01208       result_type x = Generator::Ran();
01209       if (m < nbits)
01210         b <<= (width > ulbits ? width - ulbits : width);
01211       if (width > ulbits &&     // x doesn't fit into a bitset_uint_t
01212           // But on the first time through the loop the most significant bits
01213           // may not be needed.
01214           (nbits > ((nbits-1)/width) * width + ulbits || m < nbits)) {
01215         // Handle most significant width - ulbits bits.
01216         b |= (bitset_uint_t)(x >> (width > ulbits ? ulbits : 0));
01217         b <<= ulbits;
01218       }
01219       // Bitsets can be constructed from a bitset_uint_t.
01220       b |= (bitset_uint_t)(x);
01221       m -= width;
01222     }
01223     return b;
