00472     template<typename RealType, int prec> RealType FixedC() throw() {
00473       // A real of type RealType in [0, 1] with precision prec
00474       STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
00475                     std::numeric_limits<RealType>::radix == 2,
00476                     "FixedC(): bad real type RealType");
00477       STATIC_ASSERT(prec > 0 && prec <= std::numeric_limits<RealType>::digits,
00478                     "FixedC(): invalid precision");
00479       if (prec < width) {
00480         // Sample an integer in [0, n) where n = 2^prec + 1.  This uses the
00481         // same logic as Unsigned(n - 1).  However, unlike Unsigned, there
00482         // doesn't seem to be much of a penalty for the 64-bit arithmetic here
00483         // when result_type = unsigned long long.  Presumably this is because
00484         // the compiler can do some of the arithmetic.
00485         const result_type
00486           n = (result_type(1) << (prec < width ? prec : 0)) + 1,
00487           // Computing this instead of 2^width/n suffices, because of the form
00488           // of n.
00489           r = Generator::mask / n,
00490           m = r * n;
00491         result_type u;
00492         do
00493           u = Generator::Ran();
00494         while (u >= m);
00495         // u is rv in [0, r * n)
00496         return RandomPower2::shiftf<RealType>(RealType(u / r), -prec);
00497         // Could also special case prec < 64, using Ran64().  However the
00498         // general code below is faster.
00499       } else {                  // prec >= width
00500         // Synthesize a prec+1 bit random, Y, width bits at a time.  If number
00501         // is odd, return Fixed<RealType, prec>() (w prob 1/2); else if number
00502         // is zero, return 1 (w prob 1/2^(prec+1)); else repeat.  Normalizing
00503         // probabilities on returned results we find that Fixed<RealType,
00504         // prec>() is returned with prob 2^prec/(2^prec+1), and 1 is return
00505         // with prob 1/(2^prec+1), as required.  Loop executed twice on average
00506         // and so consumes 2rvs more than rvs for Fixed<RealType, prec>().  As
00507         // in FloatZ, do NOT try to save on calls to Ran() by using the
00508         // leftover bits from Fixed.
00509         while (true) {
00510           // If prec + 1 < width then mask x with (1 << prec + 1) - 1
00511           const result_type x = Generator::Ran(); // Low width bits of Y
00512           if (x & 1u)                             // Y odd?
00513             return Fixed<RealType, prec>(); // Prob 1/2 on each loop iteration
00514           if (x)
00515             continue;               // Y nonzero
00516           int s = prec + 1 - width; // Bits left to check (s >= 0)
00517           while (true) {
00518             if (s <= 0)         // We're done.  Y = 0
00519               // Prob 1/2^(prec+1) on each loop iteration
00520               return RealType(1); // We get here once every 60000 yrs (p = 64)!
00521             // Check the next min(s, width) bits.
00522             if (Generator::Ran() >> (s > width ? 0 : width - s))
00523               break;
00524             s -= width;         // Decrement s
00525           }
00526         }
00527       }
00528     }

00868   template<class Generator> template<typename UIntT>
00869   inline typename UIntT::type
00870   RandomCanonical<Generator>::Unsigned(typename UIntT::type n) throw() {
00871     // A random unsigned in [0, n].  In n fits in 32-bits, call with UIntType =
00872     // u32 and onep = true else call with UIntType = u64 and onep = false.
00873     // There are a few cases (e.g., n = 0x80000000) where on a 64-bit machine
00874     // with a 64-bit Generator it would be quicker to call this with UIntType =
00875     // result_type and invoke Ran().  However this speed advantage disappears
00876     // if the argument isn't a compile time constant.
00877     //
00878     // Special case n == 0 is handled by the callers of Unsigned.  The
00879     // following is to guard against a division by 0 in the return statement
00880     // (but it shouldn't happen).
00881     n = n ? n : 1U;             // n >= 1
00882     // n1 = n + 1, but replace overflowed value by 1.  Overflow occurs, e.g.,
00883     // when n = u32::mask and then we have r1 = 0, m = u32::mask.
00884     const typename UIntT::type n1 = ~n ? n + 1U : 1U;
00885     // "Ratio method".  Find m = r * n1 - 1, s.t., 0 < (q - n1 ) < m <= q,
00886     // where q = max(UIntType), and sample in u in [0, m] and return u / r.  If
00887     // onep then we use Ran32() else Rand64().
00888     const typename UIntT::type
00889       // r = floor((q + 1)/n1), r1 = r - 1, avoiding overflow.  Actually
00890       // overflow can occur if std::numeric_limits<u32>::digits == 64, because
00891       // then we can have onep && n > U32_MASK.  This is however ruled out by
00892       // the callers to Unsigned.  (If Unsigned is called in this way, the
00893       // results are bogus, but there is no error condition.)
00894       r1 = ((UIntT::width == 32 ? typename UIntT::type(u32::mask) :
00895              typename UIntT::type(u64::mask)) - n) / n1,
00896       m = r1 * n1 + n;          // m = r * n1 - 1, avoiding overflow
00897     // Here r1 in [0, (q-1)/2], m in [(q+1)/2, q]
00898     typename UIntT::type u;     // Find a random number in [0, m]
00899     do
00900       // For small n1, this is executed once (since m is nearly q).  In the
00901       // worst case the loop is executed slightly less than twice on average.
00902       u = UIntT::width == 32 ? typename UIntT::type(Generator::Ran32()) :
00903         typename UIntT::type(Generator::Ran64());
00904     while (u > m);
00905     // Now u is in [0, m] = [0, r * n1), so u / r is in [0, n1) = [0, n].  An
00906     // alternative unbiased method would be u % n1; but / appears to be faster.
00907     return u / (r1 + 1U);
00908   }

00970     // usable random bits.
00971     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
00972                   std::numeric_limits<RealType>::radix == 2,
00973                   "FloatZ: bad real type RealType");
00974     STATIC_ASSERT(prec > 0 && prec <= std::numeric_limits<RealType>::digits,
00975                   "FloatZ: invalid precision");
00976     STATIC_ASSERT(erange >= 0, "FloatZ: invalid exponent range");
00977     // With subnormals: condition that smallest number is representable
00978     STATIC_ASSERT(!RANDOMLIB_HASDENORM(RealType) ||
00979                   // Need 1/2^(erange+prec) > 0
00980                   prec + erange <= std::numeric_limits<RealType>::digits -
00981                   std::numeric_limits<RealType>::min_exponent,
00982                   "FloatZ: smallest number cannot be represented");
00983     // Without subnormals :condition for no underflow in while loop
00984     STATIC_ASSERT(RANDOMLIB_HASDENORM(RealType) ||
00985                   // Need 1/2^(erange+1) > 0
00986                   erange <= - std::numeric_limits<RealType>::min_exponent,
00987                   "FloatZ: underflow possible");
00988 
00989     // Simpler (but slower) version of FloatZ.  However this method cannot
00990     // handle the full range of exponents and, in addition, is slower on
00991     // average.
00992     // template<typename RealType, int prec, int erange, bool up>
00993     // RealType FloatZ() {
00994     //   RealType x = Fixed<RealType, erange + 1>();
00995     //   int s;         // Determine exponent (-erange <= s <= 0)
00996     //   frexp(x, &s);      // Prob(s) = 2^(s-1)
00997     //   // Scale number in [1,2) by 2^(s-1).  If x == 0 scale number in [0,1).
00998     //   return ((up ? FixedU<RealType, prec - 1>() :
00999     //            Fixed<RealType, prec - 1>()) + (x ? 1 : 0)) *
01000     //     RandomPower2::pow2<RealType>(s - 1);
01001     // }
01002     //
01003     // Use {a, b} to denote the inteval: up ? (a, b] : [a, b)
01004     //
01005     // The code produces the number as
01006     //
01007     // Interval             count       prob = spacing
01008     // {1,2} / 2            2^(prec-1)  1/2^prec
01009     // {1,2} / 2^s          2^(prec-1)  1/2^(prec+s-1)      for s = 2..erange+1
01010     // {0,1} / 2^(erange+1) 2^(prec-1)  1/2^(prec+erange)
01011 
01012     // Generate prec bits in {0, 1}
01013     RealType x = up ? FixedU<RealType, prec>() : Fixed<RealType, prec>();
01014     // Use whole interval if erange == 0 and handle the interval {1/2, 1}
01015     if (erange == 0 || (up ? x > RealType(0.5) : x >= RealType(0.5)))
01016       return x;
01017     x += RealType(0.5);         // Shift remaining portion to {1/2, 1}
01018     if (b == 0) {
01019       m = Generator::Ran();     // Random bits
01020       b = width;                // Bits available in m
01021     }
01022     int sm = erange;            // sm = erange - s + 2
01023     // Here x in {1, 2} / 2, prob 1/2
01024     do {                        // s = 2 thru erange+1, sm = erange thru 1
01025       x /= 2;
01026       if (m & 1u)
01027         return x;               // x in {1, 2} / 2^s, prob 1/2^s
01028       if (--b)
01029         m >>= 1;
01030       else {
01031         m = Generator::Ran();
01032         b = width;
01033       }
01034     } while (--sm);
01035     // x in {1, 2} / 2^(erange+1), prob 1/2^(erange+1).  Don't worry about the
01036     // possible overhead of the calls to pow here.  We rarely get here.
01037     if (RANDOMLIB_HASDENORM(RealType) || // subnormals allowed
01038         // No subnormals but smallest number still representable
01039         prec + erange <= -std::numeric_limits<RealType>::min_exponent + 1 ||
01040         // Possibility of underflow, so have to test on x.  Here, we have -prec
01041         // + 1 < erange + min_exp <= 0 so pow2 can be used
01042         x >= (RealType(1) +
01043               RandomPower2::pow2<RealType>
01044               (erange + std::numeric_limits<RealType>::min_exponent)) *
01045         (erange + 1 > -RandomPower2::minpow ?
01046          std::pow(RealType(2), - erange - 1) :
01047          RandomPower2::pow2<RealType>(- erange - 1)))
01048       // shift x to {0, 1} / 2^(erange+1)
01049       // Use product of pow's since max(erange + 1) =
01050       // std::numeric_limits<RealType>::digits -
01051       // std::numeric_limits<RealType>::min_exponent and pow may underflow
01052       return x -
01053         (erange + 1 > -RandomPower2::minpow ?
01054          std::pow(RealType(2), -(erange + 1)/2) *
01055          std::pow(RealType(2), -(erange + 1) + (erange + 1)/2) :
01056          RandomPower2::pow2<RealType>(- erange - 1));
01057     else
01058       return up ?               // Underflow to up ? min() : 0
01059         // pow is OK here.
01060         std::pow(RealType(2), std::numeric_limits<RealType>::min_exponent - 1)
01061         : RealType(0);
01062   }

01075   // True with probability p.  true if p >= 1, false if p <= 0 or isnan(p).
01076   template<class Generator> template<typename RealType>
01077   inline bool RandomCanonical<Generator>::ProbF(RealType p) throw() {
01078     // Simulate Float<RealType>() < p.  The definition involves < (instead of
01079     // <=) because Float<RealType>() is in [0,1) so it is "biased downwards".
01080     // Instead of calling Float<RealType>(), we generate only as many bits as
01081     // necessary to determine the result.  This makes the routine considerably
01082     // faster than Float<RealType>() < x even for type float.  Compared with
01083     // the inexact Fixed<RealType>() < p, this is about 20% slower with floats
01084     // and 20% faster with doubles and long doubles.
01085     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
01086                   std::numeric_limits<RealType>::radix == 2,
01087                   "ProbF(p): invalid real type RealType");
01088     // Generate Real() c bits at a time where c is chosen so that cast doesn't
01089     // loose any bits and so that it uses up just one rv.
01090     const int c = std::numeric_limits<RealType>::digits > width ?
01091       width : std::numeric_limits<RealType>::digits;
01092     STATIC_ASSERT(c > 0, "ProbF(p): Illegal chunk size");
01093     const RealType mult = RandomPower2::pow2<RealType>(c);
01094     // A recursive definition:
01095     //
01096     // return p > RealType(0) &&
01097     //   (p >= RealType(1) ||
01098     //    ProbF(mult * p - RealType(Integer<result_type, c>())));
01099     //
01100     // Pre-loop tests needed to avoid overflow
01101     if (!(p > RealType(0)))     // Ensure false if isnan(p)
01102       return false;
01103     else if (p >= RealType(1))
01104       return true;
01105     do {                        // Loop executed slightly more than once.
01106       // Here p is in (0,1).  Write Fixed() = (X + y)/mult where X is an
01107       // integer in [0, mult) and y is a real in [0,1).  Then Fixed() < p
01108       // becomes p' > y where p' = p * mult - X.
01109       p *= mult;                // Form p'.  Multiplication is exact
01110       p -= RealType(Integer<result_type, c>()); // Also exact
01111       if (p <= RealType(0))
01112         return false;           // If p' <= 0 the result is definitely false.
01113       // Exit if p' >= 1; the result is definitely true.  Otherwise p' is in
01114       // (0,1) and the result is true with probability p'.
01115     } while (p < RealType(1));
01116     return true;
01117   }

01130   // True with probability x/y (ratio of reals)
01131   template<class Generator> template<typename RealType>
01132   inline bool RandomCanonical<Generator>::ProbF(RealType x, RealType y)
01133     throw() {
01134     STATIC_ASSERT(!std::numeric_limits<RealType>::is_integer &&
01135                   std::numeric_limits<RealType>::radix == 2,
01136                   "ProbF(x,y): invalid real type RealType");
01137     if (!(x > RealType(0) && y >= RealType(0))) // Do the trivial cases
01138       return false;             // Also if either x or y is a nan
01139     else if (x >= y)
01140       return true;
01141     // Now 0 < x < y
01142     int ex, ey;                 // Extract exponents
01143     x = std::frexp(x, &ex);
01144     y = std::frexp(y, &ey);
01145     // Now 0.5 <= x,y < 1
01146     if (x > y) {
01147       x *= RealType(0.5);
01148       ++ex;
01149     }
01150     int s = ey - ex;
01151     // Now 0.25 < x < y < 1, s >= 0, 0.5 < x/y <= 1
01152     // Return true with prob 2^-s * x/y
01153     while (s > 0) {             // With prob 1 - 2^-s return false
01154       // Check the next min(s, width) bits.
01155       if (Generator::Ran() >> (s > width ? 0 : width - s))
01156         return false;
01157       s -= width;
01158     }
01159     // Here with prob 2^-s
01160     const int c = std::numeric_limits<RealType>::digits > width ?
01161       width : std::numeric_limits<RealType>::digits;
01162     STATIC_ASSERT(c > 0, "ProbF(x,y): invalid chunk size");
01163     const RealType mult = RandomPower2::pow2<RealType>(c);
01164     // Generate infinite precision z = Real().
01165     // As soon as we know z > y, start again
01166     // As soon as we know z < x, return true
01167     // As soon as we know x < z < y, return false
01168     while (true) {              // Loop executed 1/x on average
01169       double xa = x, ya = y;
01170       while (true) {            // Loop executed slightly more than once
01171         // xa <= ya, ya > 0, xa < 1.
01172         // Here (xa,ya) are in (0,1).  Write Fixed() = (X + y)/mult where X is
01173         // an integer in [0, mult) and y is a real in [0,1).  Then Fixed() < z
01174         // becomes z' > y where z' = z * mult - X.
01175         const RealType d = RealType(Integer<result_type, c>());
01176         if (ya < RealType(1)) {
01177           ya *= mult;           // Form ya'
01178           ya -= d;
01179           if (ya <= RealType(0))
01180             break;              // z > y, start again
01181         }
01182         if (xa > RealType(0)) {
01183           xa *= mult;           // From xa'
01184           xa -= d;
01185           if (xa >= RealType(1))
01186             return true;        // z < x
01187         }
01188         if (xa <= RealType(0) && ya >= RealType(1))
01189           return false;         // x < z < y
01190       }
01191     }
01192   }

01194   template<class Generator> template<int nbits>
01195   inline std::bitset<nbits> RandomCanonical<Generator>::Bits() throw() {
01196     // Return nbits random bits
01197     STATIC_ASSERT(nbits >= 0, "Bits<n>(): invalid nbits");
01198     const int ulbits = std::numeric_limits<bitset_uint_t>::digits;
01199     STATIC_ASSERT(2 * ulbits >= width,
01200                   "Bits<n>(): integer constructor type too narrow");
01201     std::bitset<nbits> b;
01202     int m = nbits;
01203 
01204     while (m > 0) {
01205       result_type x = Generator::Ran();
01206       if (m < nbits)
01207         b <<= (width > ulbits ? width - ulbits : width);
01208       if (width > ulbits &&     // x doesn't fit into a bitset_uint_t
01209           // But on the first time through the loop the most significant bits
01210           // may not be needed.
01211           (nbits > ((nbits-1)/width) * width + ulbits || m < nbits)) {
01212         // Handle most significant width - ulbits bits.
01213         b |= (bitset_uint_t)(x >> (width > ulbits ? ulbits : 0));
01214         b <<= ulbits;
01215       }
01216       // Bitsets can be constructed from a bitset_uint_t.
01217       b |= (bitset_uint_t)(x);
01218       m -= width;
01219     }
01220     return b;
01221   }
