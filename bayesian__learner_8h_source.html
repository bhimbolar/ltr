00053 template <class TElement, class TDensityLearner>
00054 void BayesianLearner<TElement, TDensityLearner>::learnImpl(
00055     const DataSet<TElement>& data,
00056     BayesianScorer* scorer) {
00057   LabelToCapacity LabelsCapacity = CalculateLabelsCapacity(data);
00058   LabelToPriorProbability prior_probability;
00059   for (map<double, int>::iterator capacity_iterator = LabelsCapacity.begin();
00060        capacity_iterator != LabelsCapacity.end();
00061        ++capacity_iterator) {
00062     prior_probability[capacity_iterator->first] =
00063       ((double)capacity_iterator->second) / data.size();

00066   density_learner_.learn(data);
