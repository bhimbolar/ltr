00175 //    necessary because Microsoft Visual C++ 6.0 does not allow template
00176 //    functions to be friends of template classes.)
00177 // arguments:
00178 //    pRawPtr:   a raw pointer received from some other smart pointer

00183 :  m_pRawPtr(pRawPtr),
00184   m_pRefCount(pRefCount) {
00185   // if dynamic_cast failed or a cast on a NULL pointer is being performed

00187     m_pRefCount = NULL;
00188   } else if (m_pRefCount != NULL) {

00193     m_pRefCount = new PtrRefCounter();

00336 //  ResetPtr  [public]
00337 //
00338 // description:
00339 //    Resets the pointer. It will remove one reference from the active
00340 //    count of the Reference Counting Object (m_pRefCount). This could
00341 //    cause the raw pointer object (m_pRawPtr) and/or the Reference Counting
00342 //    Object (m_pRefCount) to get deleted.
00343 //    Also sets both internal pointers to NULL.
00344 template<class T>
00345 inline void shared_ptr<T>::ResetPtr(void) {
00346   if (m_pRefCount != NULL) {
00347     int iDeleteObject = m_pRefCount->SubtractActiveRef();
00348     if (iDeleteObject != 0) {
00349       delete m_pRawPtr;
00350 
00351       if (iDeleteObject == DELETE_REF_COUNTER) {
00352         delete m_pRefCount;
00353       }
00354     }
00355 
00356     m_pRefCount = NULL;
00357     m_pRawPtr = NULL;
00358   }
00359 }

00393 //  AdjustRefCount  [private]
00394 //
00395 // description:
00396 //    Increments the Reference Count in the Reference Counting Object.
00397 //    If this Active Pointer has been initialized with an Inactive Pointer,
00398 //    this function will handle determining if the values are valid. It is
00399 //    possible that a call to this function could set both the Raw and the

00402 inline void  shared_ptr<T>::AdjustRefCount(void) {
00403   if (m_pRefCount != NULL) {
00404     if (m_pRefCount->AddActiveRef() == false) {
00405       // if AddActiveRef() was unsucessful, then the Reference Counting
