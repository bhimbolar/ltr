00079 static inline void GenericParse(const TagHandlers& parsers,
00080                                 TiXmlNode* xmlNode,
00081                                 TagHandler* on_unknown_token = NULL) {
00082   for (; xmlNode; xmlNode = xmlNode->NextSibling()) {
00083     if (xmlNode->Type() != TiXmlNode::TINYXML_ELEMENT) {
00084       continue;
00085     }
00086 
00087     TiXmlElement* element = xmlNode->ToElement();
00088     if (!element) {
00089       throw logic_error("can not convert node to element");
00090     }
00091 
00092     const TagHandlers::const_iterator it = parsers.find(xmlNode->Value());
00093     if (it == parsers.end()) {
00094       if (on_unknown_token) {
00095         (*on_unknown_token)(element);
00096       } else {
00097         throw logic_error(string("Unknown token Value: ") +
00098                           xmlNode->Value() + ", Type: " +
00099                           ltr::utility::lexical_cast<string>(xmlNode->Type()) +
00100                           "\n");
00101       }
00102     } else {
00103       (*it->second)(element);
00104     }
00105   }
00106 }
