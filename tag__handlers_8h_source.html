00072 static inline void GenericParse(const TagHandlers& parsers,
00073                                 TiXmlNode* xmlNode,
00074                                 TagHandler* on_unknown_token = NULL) {
00075   for (; xmlNode; xmlNode = xmlNode->NextSibling()) {
00076     if (xmlNode->Type() != TiXmlNode::TINYXML_ELEMENT) {
00077       continue;
00078     }
00079 
00080     TiXmlElement* element = xmlNode->ToElement();
00081     if (!element) {
00082       throw logic_error("Cannot convert node to element");
00083     }
00084 
00085     const TagHandlers::const_iterator it = parsers.find(xmlNode->Value());
00086     if (it == parsers.end()) {
00087       if (on_unknown_token) {
00088         (*on_unknown_token)(element);
00089       } else {
00090         throw logic_error(string("Unknown token Value: ") +
00091                           xmlNode->Value() + ", Type: " +
00092                           boost::lexical_cast<string>(xmlNode->Type()) +
00093                           "\n");
00094       }
00095     } else {
00096       (*it->second)(element);
00097     }
00098   }
00099 }

00299     rInfo("TOnParametersExecutor: guessing type of %s %d\n",
