00078 static inline void GenericParse(const TagHandlers& parsers,
00079                                 TiXmlNode* xmlNode,
00080                                 TagHandler* on_unknown_token = NULL) {
00081   for (; xmlNode; xmlNode = xmlNode->NextSibling()) {
00082     if (xmlNode->Type() != TiXmlNode::TINYXML_ELEMENT) {
00083       continue;
00084     }
00085 
00086     TiXmlElement* element = xmlNode->ToElement();
00087     if (!element) {
00088       throw logic_error("can not convert node to element");
00089     }
00090 
00091     const TagHandlers::const_iterator it = parsers.find(xmlNode->Value());
00092     if (it == parsers.end()) {
00093       if (on_unknown_token) {
00094         (*on_unknown_token)(element);
00095       } else {
00096         throw logic_error(string("Unknown token Value: ") +
00097                           xmlNode->Value() + ", Type: " +
00098                           ltr::utility::lexical_cast<string>(xmlNode->Type()) +
00099                           "\n");
00100       }
00101     } else {
00102       (*it->second)(element);
00103     }
00104   }
00105 }
