00100 template<class TElement>
00101 Vertex::Ptr DecisionTreeLearner<TElement>::
00102   buildTree(const DataSet<TElement>& data) {
00103   if (data.size() == 0) {
00104     rError("Data is empty.");
00105     throw std::logic_error("no objects given to decision tree learner");
00106   }
00107 
00108   DataSet<Object> object_data;
00109   for (int element_index = 0;
00110        element_index < data.size(); ++element_index) {
00111        PerObjectAccessor<const TElement>
00112           per_object_accessor(&data[element_index]);
00113         for (int object_index = 0;
00114              object_index < per_object_accessor.object_count();
00115              ++object_index) {
00116               object_data.add(per_object_accessor.object(object_index));
00117         }
00118   }
00119 
00120   Vertex::Ptr root = new DecisionVertex;
00121   vector<VertexWithData> current_layer;
00122   current_layer.push_back(VertexWithData(root, object_data));
00123 
00124   while (!current_layer.empty()) {
00125     vector<VertexWithData> need_to_split, next_layer, vertex_split_result;
00126     for (int vertex_index = 0;
00127        vertex_index < (int)current_layer.size(); ++vertex_index) {
00128       if (needToStopSplitting(current_layer[vertex_index].data)) {
00129         LeafVertex::Ptr leaf = generateLeaf(current_layer[vertex_index].data);
00130         current_layer[vertex_index].vertex->addChild(leaf, new FakeCondition());
00131       } else if (tryToSplit(current_layer[vertex_index], vertex_split_result)) {
00132        // if vertex can't be splitted, it should become a leaf
00133         LeafVertex::Ptr leaf = generateLeaf(current_layer[vertex_index].data);
00134         current_layer[vertex_index].vertex->addChild(leaf, new FakeCondition());
00135       } else {  // we should split the vertex
00136         next_layer.insert(next_layer.end(),
00137                           vertex_split_result.begin(),
00138                           vertex_split_result.end());
00139       }
00140     }
00141     if (next_layer.empty()) {
00142       break;
00143     }
00144     swap(current_layer, next_layer);
00145   }
00146   return root;
00147 }
