00223     template<class Random> static int Choose(Random& r, int k) {
00224       // Limit base to 2^15 to avoid integer overflow
00225       const int b = bits > 15 ? 15 : bits;
00226       const unsigned mask = (1u << b) - 1;
00227       const int m = 2 * k + 2;
00228       int n1 = m - 2, n2 = m - 1;
00229       // Evaluate u < n/m where u is a random real number in [0,1).  Write u =
00230       // (d + u') / 2^b where d is a random integer in [0,2^b) and u' is in
00231       // [0,1).  Then u < n/m becomes u' < n'/m where n' = 2^b * n - d * m and
00232       // exit if n' <= 0 (false) or n >= m (true).
00233       while (true) {
00234         int d = (mask & RandomNumber<bits>::RandomDigit(r)) * m;
00235         n1 = std::max((n1 << b) - d, 0);
00236         if (n1 >= m) return 1;
00237         n2 = std::min((n2 << b) - d, m);
00238         if (n2 <= 0) return -1;
00239         if (n1 == 0 && n2 == m) return 0;

00255     // b = mean bit count for true  result = 2.236
00256     //
00257     // for bits large
00258     //   t = exp(1/2) = 1.6487
00259     //   a = exp(1/2)/(2*(1-exp(-1/2))) = 2.0951
00260     //   b = exp(1/2)/(2*exp(-1/2)) = 1.3591
00261     //
00262     // Results for Prob(exp(-1)), omitting first test
00263     // total = 5.889, false = 5.347, true = 6.826
00264     //
00265     // Results for Prob(exp(-1)) using ExpProbH(r) && ExpProbH(r),
00266     // total = 4.572 = (1 - exp(-1)) * a + (1 + exp(-1/2)) * exp(-1/2) * b
