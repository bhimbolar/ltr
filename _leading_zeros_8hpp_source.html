00051     // It's simpler to count the number of trailing ones in each w-bit block
00052     // stopping when we get to a zero bit.
00053     //
00054     // Process a word in chunks of size m.  The algorithm here can deal with
00055     // any m assuming that z is modified accordingly.  m = 4 is an approximate
00056     // optimum.
00057     //
00058     // Can also adapt this routine to use RandomNumber::highest_bit_idx
00059     // instead.  However the result is considerably slower.
00060     const int m = 4;
00061     STATIC_ASSERT(m <= Random::width, "LeadingZeros: m too large");
00062     // mask with m low bits set
00063     const typename Random::result_type mask = ~(Random::max << m);
00064     // Number of trailing 1 bits in [0, 1<<m).  However, correct results are
00065     // also obtained with any permutation of this array.  This particular
00066     // permutation is useful since the initial 1/2, 1/4, etc. can be used for
00067     // m-1, m-2, etc.  To generate the array for the next higher m, append a
00068     // duplicate of the array and increment the last entry by one.
00069     const unsigned z[1 << m] =
00070       { 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, };
00071     typename Random::result_type x = r();
00072     for (unsigned b = m, n = 0; b < Random::width; b += m) {
00073       n += z[x & mask];         // count trailing 1s in chunk
00074       if (n < b)                // chunk contains a 0
00075         return n;
00076       x >>= m;                  // shift out the chunk we've processed
00077     }
00078     // x is all ones (prob 1/2^w); process the next word.
00079     return Random::width + operator()(r);
00080   }
